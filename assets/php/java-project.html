    private void BotonGuardarActionPerformed(java.awt.event.ActionEvent evt) {                                             
        // TODO add your handling code here:
        String Apellidos, Nombres, ECivil;
        float Horas, Precio, SBruto, SNP, Bonificacion, SNeto;
        
        try {
            Apellidos = txtApellidos.getText();
            Nombres = txtNombres.getText();

            Horas = Float.parseFloat(txtHoras.getText());
            Precio = Float.parseFloat(txtPrecio.getText());

            SBruto = Horas * Precio;
            SNP = (float) (SBruto * 0.13);
            if (txtECivil.isSelected()) {
                Bonificacion = (float) (SBruto * 0.07);
                ECivil = "CASADO";
                showBono.setForeground(Color.blue);
                showBono.setText(String.valueOf(Bonificacion));
            } else {
                Bonificacion = 0;
                ECivil = "SOLTERO";
                showBono.setForeground(Color.black);
                showBono.setText("No tiene bonificación");
            }

            SNeto = SBruto - SNP + Bonificacion;

            String NCompleto = Apellidos.trim() + " " + Nombres.trim();
            showNombres.setText(NCompleto);
            showECivil.setText(ECivil);
            showHoras.setText(String.valueOf(Horas));
            showPrecio.setText(String.valueOf(Precio));
            showSBruto.setText(String.valueOf(SBruto));
            showSNP.setText(String.valueOf(SNP));
            showSNeto.setText(String.valueOf(SNeto));
            
        } catch (Exception e) {
            JOptionPane.showMessageDialog(null, "¡Revise que los datos estén llenados correctamente!");
        }
    }

        showPrecio.setText("");
        showSBruto.setText("");
        showSNP.setText("");
        showBono.setText("");
        showSNeto.setText("");
            
    }
